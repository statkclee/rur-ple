def turn_right():
    repeat(turn_left, 3)

put_beeper()
# begin changes
if not front_is_clear():
    turn_left()
# end changes
move()
while not on_beeper():
    if right_is_clear():
        turn_right()
        move()
    elif front_is_clear():
        move()
    else:
        turn_left()

turn_off()
